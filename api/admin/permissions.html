# Permissions

# List permissions

Get permission list

# Request

GET /api/admin/permissions
    ?search={searchText}
    &page={page}
    &limit={limit}

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

200 OK

{
    "data": [{...}],
    "meta": {...}
}

# Create permission

Create new permission entry

# Request

POST /api/admin/permissions

Authorization: Bearer {accessToken}
Content-Type: application/json

{
    "name": string,
    "displayName": string,
    "description": string
}

# Response

201 Created

{
    "data": {...}
}

# Get permission

Get permission entry

# Request

GET /api/admin/permissions/:permissionId

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

200 OK

{
    "data": {...},
    "refs": {...}
}

# Update permission

Update permission entry

# Request

PUT /api/admin/permissions/:permissionId

Authorization: Bearer {accessToken}
Content-Type: application/json

{
    "name": string,
    "displayName": string,
    "description": string
}

# Response

200 OK

{
    "data": {...},
    "refs": {...}
}

# Delete permission

Delete permission entry

# Request

DELETE /api/admin/permissions/:permissionId

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

204 No Content
