# Image maps

# Browse image maps

Browse paginated image map list

# Request

GET /api/admin/images/maps
    ?search={searchText}
    &page={page}
    &limit={limit}

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

200 OK

{
    "data": [{...}],
    "meta": {...}
}

# Create image map

Create new image map entry

  • use multipart/form-data data encoding to image file needs to be uploaded.

# Request

POST /api/admin/images/maps

Authorization: Bearer {accessToken}
Content-Type: application/json

{
    "id": string,
    "description": string,
    "baseImage": File
}

# Response

201 Created

{
    "data": {...}
}

# Get image map

Get image map entry

# Request

GET /api/admin/images/maps/:imageMapId

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

200 OK

{
    "data": {...},
    "refs": {...}
}

# Update image map

Update image map entry

# Request

PUT /api/admin/images/maps/:imageMapId

Authorization: Bearer {accessToken}
Content-Type: application/json

{
    "description": string,
    "objects": [
        {
            "id": number,
            "description": string,
            "outlineCoordinates": number[]
        },
        ...
    ]
}

# Response

200 OK

{
    "data": {...},
    "refs": {...}
}

# Delete image map

Delete image map entry

# Request

DELETE /api/admin/images/maps/:imageMapId

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

204 No Content
