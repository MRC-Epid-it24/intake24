# Roles

# Browse roles

Browse paginated role list

# Request

GET /api/admin/roles
    ?search={searchText}
    &page={page}
    &limit={limit}

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

200 OK

{
    "data": [{...}],
    "meta": {...}
}

# Create role

Create new role entry

# Request

POST /api/admin/roles

Authorization: Bearer {accessToken}
Content-Type: application/json

{
    "name": string,
    "displayName": string,
    "description": string,
    "permissions": number[]
}

# Response

201 Created

{
    "data": {...}
}

# Get role

Get role entry

# Request

GET /api/admin/roles/:roleId

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

200 OK

{
    "data": {...},
    "refs": {
        "permissions": [{...}]
    }
}

# Update role

Update role entry

# Request

PUT /api/admin/roles/:roleId

Authorization: Bearer {accessToken}
Content-Type: application/json

{
    "name": string,
    "displayName": string,
    "description": string,
    "permissions": number[]
}

# Response

200 OK

{
    "data": {...},
    "refs": {
        "permissions": [{...}]
    }
}

# Delete role

Delete role entry

# Request

DELETE /api/admin/roles/:roleId

Authorization: Bearer {accessToken}
Content-Type: application/json

# Response

204 No Content
